Accès eau potable % moyen =
AVERAGE(Fact_KPI_Monde[acces_eau_potable_pct])

Taux accès eau % - affichage =
VAR v = [Accès eau potable % moyen]
RETURN
IF(ISBLANK(v), "ND", FORMAT(v, "0.00"))

Couleur accès eau mixte DWFA =
VAR v = [Accès eau potable % moyen]
RETURN
SWITCH(
    TRUE(),
    ISBLANK(v), "#6E8597",
    v < 50, "#B10026",
    v < 65, "#FC4E2A",
    v < 75, "#FEB24C",
    v < 85, "#8CCCED",
    v < 92, "#2F9AD6",
    "#163A59"
)

Taux mortalité WASH moyen =
AVERAGE(Fact_KPI_Monde[mortalite_wash_pour_100k])

Badge alerte mortalité - texte =
VAR m = [Taux mortalité WASH moyen]
RETURN
SWITCH(
    TRUE(),
    ISBLANK(m), "ND",
    m >= 20, "Situation défavorable",
    m >= 5, "Situation à surveiller",
    "Situation favorable"
)

Badge alerte mortalité - couleur =
VAR m = [Taux mortalité WASH moyen]
RETURN
SWITCH(
    TRUE(),
    ISBLANK(m), "#6E8597",
    m >= 20, "#B10026",
    m >= 5, "#FEB24C",
    "#2F9AD6"
)

Afficher_Pays_Eligibles_v8 =
VAR iso = SELECTEDVALUE(dim_pays_star_fr[iso3])
VAR choixPop = SELECTEDVALUE(dim_pays_star_fr[Eligible_pop_500k], "Oui (>500k)")
VAR choixStab =
IF(
    HASONEVALUE('Filtre_stabilité_politique'[Option]),
    SELECTEDVALUE('Filtre_stabilité_politique'[Option]),
    "Inclure les pays instables"
)
VAR choixEff =
IF(
    HASONEVALUE(Filtre_effectivite_gouvernementale[Option]),
    SELECTEDVALUE(Filtre_effectivite_gouvernementale[Option]),
    "Inclure les pays performants"
)
VAR seuil1 = [Seuil stabilité]
VAR stab = [Stabilité politique moyenne KPI]
VAR seuil2 = [Seuil effectivité sélectionné]
VAR eff = [Score effectivite gouvernementale exacte]
VAR popPays = SELECTEDVALUE(dim_pays_star_fr[Eligible_pop_500k], "Inconnu")
VAR okGeo =
NOT ISBLANK(iso) &&
NOT ( iso IN { "HKG", "MAC", "TWN" } )
VAR okPop =
SWITCH(
    choixPop,
    "Oui (>500k)", popPays = "Oui (>500k)",
    "Non (<=500k)", popPays = "Non (<=500k)",
    "Inconnu", popPays = "Inconnu",
    TRUE()
)
VAR okStab =
IF(
    choixStab = "Exclure les pays instables",
    NOT ISBLANK(stab) && stab >= seuil1,
    TRUE()
)
VAR okEff =
IF(
    choixEff = "Exclure les pays performants",
    NOT ISBLANK(eff) && eff <= seuil2,
    TRUE()
)
RETURN IF(okGeo && okPop && okStab && okEff, 1, 0)
